// ==== CONFIGURATION ====
var API_URL = "https://majic-clip-tagger.vercel.app/api/mappings";
var API_PASSWORD = "CHANGE_ME"; // Must match APP_PASSWORD in Vercel env vars
var SOURCE_FOLDER_ID = "1OKbt3ThRACnSCW2jWxzfZcOQ4wHh1lxH";
var LOG_SHEET_ID = "1xaKOA5QPDInchMhvCK2BNkLAVPfyzpHm46LaFSQUtV0";
var LOG_FOLDER_ID = "1SrW4XWRdlsha690dZCZTc1w6tBCbmIoX";

// ==== WEB APP ====
function doGet() {
  var results = renameClips();
  var html = buildResultsPage(results);
  return HtmlService.createHtmlOutput(html)
    .setTitle("Hockey Clip Renamer");
}

function buildResultsPage(results) {
  if (results.length === 0) {
    return "<html><body style='font-family:sans-serif;padding:24px;'>"
      + "<h2>Nothing to process</h2>"
      + "<p>No submitted games found. Mary needs to tag clips and tap Submit first.</p>"
      + "</body></html>";
  }

  var rows = "";
  for (var i = 0; i < results.length; i++) {
    var r = results[i];
    var color = r.complete ? "#22c55e" : "#f59e0b";
    var status = r.complete
      ? r.processed + " of " + r.total + " - Done!"
      : r.processed + " of " + r.total + " - " + r.missing + " missing, will retry next run";
    rows += "<tr>"
      + "<td style='padding:8px;border-bottom:1px solid #eee;'>" + r.game + "</td>"
      + "<td style='padding:8px;border-bottom:1px solid #eee;'><span style='color:" + color + ";font-weight:bold;'>" + status + "</span></td>"
      + "</tr>";
  }

  return "<html><body style='font-family:sans-serif;padding:24px;'>"
    + "<h2>Done!</h2>"
    + "<table style='border-collapse:collapse;width:100%;'>"
    + "<tr><th style='padding:8px;text-align:left;border-bottom:2px solid #ddd;'>Game</th>"
    + "<th style='padding:8px;text-align:left;border-bottom:2px solid #ddd;'>Files</th></tr>"
    + rows
    + "</table>"
    + "<p style='margin-top:16px;color:#888;'>Tap to run again anytime. Partial games will be retried.</p>"
    + "</body></html>";
}

// ==== MAIN ====
function renameClips() {
  // Fetch unprocessed games from the web app API
  var response = UrlFetchApp.fetch(API_URL, {
    headers: { "authorization": API_PASSWORD },
    muteHttpExceptions: true
  });

  if (response.getResponseCode() !== 200) {
    Logger.log("ERROR: API returned " + response.getResponseCode());
    return [];
  }

  var games = JSON.parse(response.getContentText());

  // Only process games Mary has submitted
  var submitted = [];
  for (var i = 0; i < games.length; i++) {
    if (games[i].submitted) submitted.push(games[i]);
  }

  if (submitted.length === 0) {
    Logger.log("No submitted games to process.");
    return [];
  }

  Logger.log("Found " + submitted.length + " submitted game(s).");

  var sourceFolder = DriveApp.getFolderById(SOURCE_FOLDER_ID);
  var results = [];

  for (var i = 0; i < submitted.length; i++) {
    results.push(processGame(submitted[i], sourceFolder));
  }

  return results;
}

function processGame(game, sourceFolder) {
  var description = game.description;
  var mappings = game.mappings;
  var filenames = Object.keys(mappings).sort();

  var gameFolder = getOrCreateSubfolder(sourceFolder, description);

  Logger.log("Game: " + description);
  Logger.log("Clips: " + filenames.length);
  Logger.log("--------------------------------------------------");

  var renamed = 0;
  var alreadyDone = 0;
  var missing = 0;

  for (var i = 0; i < filenames.length; i++) {
    var originalName = filenames[i];
    var ext = getExtension(originalName);
    var newName = buildFilename(mappings[originalName], originalName, ext);

    // Check if already renamed in a previous run
    if (findFile(gameFolder, newName)) {
      Logger.log("  ALREADY DONE: " + newName);
      alreadyDone++;
      continue;
    }

    // Try to find the original file in the uploads folder
    var file = findFile(sourceFolder, originalName);
    if (!file) {
      Logger.log("  MISSING: " + originalName);
      missing++;
      continue;
    }

    file.setName(newName);
    file.moveTo(gameFolder);
    Logger.log("  " + originalName + " -> " + newName);
    renamed++;
  }

  var total = filenames.length;
  var processed = renamed + alreadyDone;
  var allDone = (missing === 0);

  // Log to spreadsheet
  var now = new Date();
  var sheet = SpreadsheetApp.openById(LOG_SHEET_ID).getActiveSheet();
  sheet.appendRow([
    Utilities.formatDate(now, Session.getScriptTimeZone(), "yyyy-MM-dd"),
    Utilities.formatDate(now, Session.getScriptTimeZone(), "HH:mm:ss"),
    description,
    allDone ? "Complete" : "Partial",
    processed + " of " + total,
    renamed + " new, " + alreadyDone + " already done, " + missing + " missing"
  ]);

  if (allDone) {
    // All files found and renamed - move game folder out of uploads
    var parents = sourceFolder.getParents();
    if (parents.hasNext()) {
      var marysFolder = parents.next();
      gameFolder.moveTo(marysFolder);
      Logger.log("  Moved folder to: " + marysFolder.getName() + "/" + description + "/");
    }

    // Mark game as processed in the API so it won't appear next run
    game.processed = true;
    UrlFetchApp.fetch(API_URL, {
      method: "post",
      contentType: "application/json",
      headers: { "authorization": API_PASSWORD },
      payload: JSON.stringify(game),
      muteHttpExceptions: true
    });
    Logger.log("  Marked as processed in API.");
  } else {
    Logger.log("  " + missing + " file(s) missing - will retry next run.");
  }

  Logger.log("Done: " + renamed + " renamed, " + alreadyDone + " already done, " + missing + " missing");

  return {
    game: description,
    renamed: renamed,
    processed: processed,
    missing: missing,
    total: total,
    complete: allDone
  };
}

function buildFilename(mapping, originalName, ext) {
  var cameraMatch = originalName.match(/(IMG_\d+|PXL_\d{8}_\d+|VID_\d{8}_\d+)/);
  var imgId = cameraMatch ? cameraMatch[1] : originalName.replace(/\.[^.]+$/, "");

  var parts = [];

  var line = mapping.line;
  var player = mapping.player;
  var tag = mapping.tag;
  var custom = mapping.custom;

  if (line) parts.push(line + " Line");
  if (player) parts.push(player);
  if (tag) {
    parts.push(tag);
  } else if (!line && !player) {
    parts.push("Clip");
  }
  if (custom) parts.push(custom);

  if (parts.length === 0) parts.push("Clip");

  parts.push(imgId);
  return parts.join(" ") + ext;
}

function findFile(folder, name) {
  var files = folder.getFilesByName(name);
  return files.hasNext() ? files.next() : null;
}

function getOrCreateSubfolder(parent, name) {
  var folders = parent.getFoldersByName(name);
  if (folders.hasNext()) return folders.next();
  return parent.createFolder(name);
}

function getExtension(filename) {
  var dot = filename.lastIndexOf(".");
  return dot >= 0 ? filename.substring(dot) : "";
}
